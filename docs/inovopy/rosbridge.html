357    @contextmanager
358    def jog(self):
359        """
360        Enter a context of jog command mode. Advertise when enter context
361        and automatically unadvertise when exit.
362
363        ## Yields:
364            `JogPublisher`: api for jog command
365
366        ## Example:
367        ```python
368        import time
369        from inovopy.rosbridge import InovoRos
370
371        ros = InovoRos("192.168.1.122", "Jog")
372
373        with ros.jog() as j:
374            for _ in range(1000):
375                j.jog(z=-0.01)
376                time.sleep(0.01)
377            for _ in range(1000):
378                j.jog(z=0.01)
379                time.sleep(0.01)
380        ```
381        """
382
383        top = roslibpy.Topic(self.ros, TOPIC_CARTESIAN_JOG, TYPE_CARTESIAN_JOG_DEMAND)
384        top.advertise()
385
386        publisher = JogPublisher(top)
387
388        yield publisher
389
390        publisher.publisher.unadvertise()
391        del publisher

422    def jog(
423        self,
424        x: float = 0,
425        y: float = 0,
426        z: float = 0,
427        rx: float = 0,
428        ry: float = 0,
429        rz: float = 0,
