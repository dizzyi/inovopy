 27def rx_mat(deg: float) -> numpy_typing.NDArray[numpy.float64]:
 28    """compute rotation matrx of rotation along x axis"""
 29    mat = numpy.eye(3)
 30    mat[1, 1] = numpy.cos(deg_to_rad(deg))
 31    mat[1, 2] = -numpy.sin(deg_to_rad(deg))
 32    mat[2, 2] = numpy.cos(deg_to_rad(deg))
 33    mat[2, 1] = numpy.sin(deg_to_rad(deg))
 34    return mat

 37def ry_mat(deg: float) -> numpy_typing.NDArray[numpy.float64]:
 38    """compute rotation matrx of rotation along y axis"""
 39    mat = numpy.eye(3)
 40    mat[0, 0] = numpy.cos(deg_to_rad(deg))
 41    mat[0, 2] = numpy.sin(deg_to_rad(deg))
 42    mat[2, 2] = numpy.cos(deg_to_rad(deg))
 43    mat[2, 0] = -numpy.sin(deg_to_rad(deg))
 44    return mat

 47def rz_mat(deg: float) -> numpy_typing.NDArray[numpy.float64]:
 48    """compute rotation matrx of rotation along z axis"""
 49    mat = numpy.eye(3)

 57def euler_to_mat(
 58    euler_deg: Tuple[float, float, float],
 59) -> numpy_typing.NDArray[numpy.float64]:
 60    """translate euler angle to rotation matrix"""
 61    rx = rx_mat(euler_deg[0])
 62    ry = ry_mat(euler_deg[1])
 63    rz = rz_mat(euler_deg[2])
 64    return numpy.matmul(rz, numpy.matmul(ry, rx))

 67def mat_to_euler(
 68    mat: numpy_typing.NDArray[numpy.float64],
 69) -> Tuple[float, float, float]:
 70    """translate rotation matrix to euler angle"""
 71    r31 = mat[2, 0]
 72    if numpy.abs(r31) != 1:
 73        ry = -numpy.arcsin(r31)
 74        rx = numpy.arctan2(mat[2, 1] / numpy.cos(ry), mat[2, 2] / numpy.cos(ry))
 75        rz = numpy.arctan2(mat[1, 0] / numpy.cos(ry), mat[0, 0] / numpy.cos(ry))
 76    else:
 77        rz = 0
 78        if r31 == -1:
 79            ry = numpy.pi / 2
 80            rx = numpy.arctan2(mat[0, 1], mat[0, 2])
 81        else:
 82            ry = -numpy.pi / 2
 83            rx = numpy.arctan2(-mat[0, 1], -mat[0, 2])
 84    return (rad_to_deg(rx), rad_to_deg(ry), rad_to_deg(rz))
