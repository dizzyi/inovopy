132class RobotCommand:
133    """
134    # RobotCommand
135    a class for constructing robot command
136
137    ## Class Method
138    - `synchronize` : command to break blending
139    - `sleep` : command to robot to sleep
140    - `set_parameter` : command to set motion parameter of robot
141    - `motion` : command to move the robot
142    """
143
144    def __init__(self, argument: dict[str, str | float]):
145        self.argument: dict[str, str | float] = argument
146
147    def to_dict(self) -> dict[str, str | float]:
148        """return `dict[str,str|float]` representation of the command"""
149        return self.argument
150
151    @classmethod
152    def synchronize(cls) -> "RobotCommand":
153        """Return a `RobotCommand` for synchronizing/breaking blending"""
154        return RobotCommand(
155            {
156                "action": "synchronize",
157            }
158        )
159
160    @classmethod
161    def sleep(cls, second: float) -> "RobotCommand":
162        """
163        Return a `RobotCommand` for the robot to sleep
164
165        ## Parameter
166        - `second : float` : specify the lenght of sleep in seconds
167        """
168        return RobotCommand({"action": "sleep", "second": second})
169
170    @classmethod
171    def set_parameter(
172        cls,
173        speed: float | None = None,
174        accel: float | None = None,
175        blend_linear: float | None = None,
176        blend_angular: float | None = None,
177        tcp_speed_linear: float | None = None,
178        tcp_speed_angular: float | None = None,
179    ) -> "RobotCommand":
180        """
181        Return a `RobotCommand` for setting the robot motion parameter
182
183        ## Parameter
184        - `speed : float`, in percent, range from `1` to `100`
185        - `accel : float`, in precent, range from `1` to `100`
186        - `blend_linear : float`, in mm, range from `1` to `1000`
187        - `blend_angular : float`, in degree, range from `1` to `360`
188        - `tcp_speed_linear : float`, in mm, range from `1` to `999`
189        - `tcp_speed_angular : float`, in degree, range from `1` to `360`
190        """
191
192        speed = clamp(speed / 100, 0.01, 1) if speed else 0
193
194        accel = clamp(accel / 100, 0.01, 1) if accel else 0
195
196        blend_linear = clamp(blend_linear / 1000, 0.001, 1) if blend_linear else 0
197
198        blend_angular = (
199            clamp(
200                inovopy.geometry.transform.deg_to_rad(blend_angular),
201                0.001,
202                2 * numpy.pi,
203            )
204            if blend_angular
205            else 0
206        )
207
208        tcp_speed_linear = (
209            clamp(tcp_speed_linear / 1000, 0.001, 0.999) if tcp_speed_linear else 0
210        )
211
212        tcp_speed_angular = (
213            clamp(
214                inovopy.geometry.transform.deg_to_rad(tcp_speed_angular),
215                0.001,
216                2 * numpy.pi,
217            )
218            if tcp_speed_angular
219            else 0
220        )
221
222        return RobotCommand(
223            {
224                "action": "set_parameter",
225                "speed": speed,
226                "accel": accel,
227                "blend_linear": blend_linear,
228                "blend_angular": blend_angular,
229                "tcp_speed_linear": tcp_speed_linear,
230                "tcp_speed_angular": tcp_speed_angular,
231            }
232        )
233
234    @classmethod
235    def motion(
236        cls,
237        motion_mode: "MotionMode",
238        target: "inovopy.geometry.jointcoord.JointCoord | inovopy.geometry.transform.Transform",
239    ) -> "RobotCommand":
240        """
241        Return a `RobotCommand` for robot motion
242
243        ## Parameter
244        - `motion_mode : MotionMode`, specify the motion mode
245        - `target` : target of the motion, either transform or joint coord
246        """
247        return RobotCommand(
248            {
249                "action": "motion",
250                "motion_mode": motion_mode.to_arg(),
251                **target.to_dict(),
252            }
253        )

256class MotionMode(str, Enum):
257    """
258    # MotionMode
259    An str-enum class for different robot motion's
260    interpolation mode and specify whether is relative move
261
262    ## Enum
263    - `LINEAR`
264    - `LINEAR_RELATIVE`
265    - `JOINT`
266    - `JOINT_RELATIVE`
267    """
268
269    LINEAR = "linear"
270    LINEAR_RELATIVE = "linear_relative"
271    JOINT = "joint"
272    JOINT_RELATIVE = "joint_relatve"
273
274    def to_arg(self) -> str:
275        return str(self).split(".")[1].lower()
